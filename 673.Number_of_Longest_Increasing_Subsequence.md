## Method 1: Dynamic Programming
```java 
class Solution {
    public int findNumberOfLIS(int[] nums) {
        int[] lengths = new int[nums.length];
        int[] count = new int[nums.length];
        Arrays.fill(count, 1);
        for(int j=0; j<nums.length; j++) {
            for(int i=0; i<j; i++) {
                if(nums[i] < nums[j]) {
                    if(lengths[i] >= lengths[j]) {      // append nums[j]
                        lengths[j] = lengths[i] + 1;
                        count[j] = count[i];
                    } else if(lengths[i] + 1 == lengths[j]) {
                        count[j] += count[i];
                    }
                }
            }
        }
        
        int longest = 0;
        for(int length: lengths) {
            longest = Math.max(longest, length);
        }
        int res = 0;
        for(int i=0; i<count.length; i++) {
            if(lengths[i] == longest) {
                res += count[i];
            }
        }
        return res;
    }
}
```

## Method 2: Segment Tree
```java 
class Solution {
    public int findNumberOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        Node[] f = new Node[nums.length];
        int len = 0;
        for(int i=0; i<nums.length; i++) {
            int idx = Arrays.binarySearch(dp, 0, len, nums[i]);
            if(idx < 0) {   // nums[i] is the smallest num
                idx = -(idx + 1);
            }
            dp[idx] = nums[i];
            len = Math.max(idx + 1, len);
            f[i] = new Node(idx + 1, 0);
            if(idx == 0) {
                f[i].count = 1;
            } else {
                for(int j=0; j<i; j++) {
                    if(f[j].len + 1 == f[i].len && nums[j] < nums[i]) {
                        f[i].count += f[j].count;
                    }
                }
            }
        }
        
        int res = 0;
        for(Node node: f) {
            if(node.len == len) {
                res += node.count;
            }
        }
        return res;
    }
}

class Node {
    int len;
    int count;
    
    public Node(int len, int count) {
        this.len = len;
        this.count = count;
    }
}
```
